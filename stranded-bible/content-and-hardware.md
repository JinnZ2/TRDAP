// stranded_bible_oled.ino
// For low-power Arduino version

#include <SPI.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <SD.h>

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

#define SD_CS 4
#define BUTTON_UP 2
#define BUTTON_DOWN 3
#define BUTTON_SELECT 5
#define BUZZER 6

File textFile;
String currentText = "";
int scrollPosition = 0;
int maxLines = 0;

// Menu structure
struct MenuItem {
  const char* name;
  const char* filename;
};

MenuItem mainMenu[] = {
  {"📖 Scriptures", "/bible.txt"},
  {"🙏 Prayers", "/prayers.txt"},
  {"🆘 Survival", "/survival.txt"},
  {"📝 Journal", "/journal.txt"},
  {"✍️ Last Words", "/lastwords.txt"},
  {"⚡ Power Save", ""}

int menuIndex = 0;

void setup() {
  Serial.begin(9600);
  
  // Initialize display
  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("Display failed");
    while(1);
  }
  
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  
  // Initialize SD card
  if(!SD.begin(SD_CS)) {
    display.println("No SD card");
    display.display();
    delay(2000);
  }
  
  // Initialize buttons
  pinMode(BUTTON_UP, INPUT_PULLUP);
  pinMode(BUTTON_DOWN, INPUT_PULLUP);
  pinMode(BUTTON_SELECT, INPUT_PULLUP);
  pinMode(BUZZER, OUTPUT);
  
  // Show startup
  display.clearDisplay();
  display.setCursor(0,0);
  display.println("Stranded Bible");
  display.println("v1.0");
  display.println("Press SELECT");
  display.display();
  
  // Beep once to show alive
  tone(BUZZER, 1000, 100);
}

void loop() {
  drawMenu();
  
  if(digitalRead(BUTTON_UP) == LOW) {
    menuIndex--;
    if(menuIndex < 0) menuIndex = 5;
    delay(200);
  }
  
  if(digitalRead(BUTTON_DOWN) == LOW) {
    menuIndex++;
    if(menuIndex > 5) menuIndex = 0;
    delay(200);
  }
  
  if(digitalRead(BUTTON_SELECT) == LOW) {
    if(menuIndex == 5) {
      // Power save
      display.clearDisplay();
      display.setCursor(0,0);
      display.println("Powering down...");
      display.display();
      delay(1000);
      // Enter deep sleep
      // Would need external wake circuit
    } else {
      openFile(mainMenu[menuIndex].filename);
    }
    delay(300);
  }
}

void drawMenu() {
  display.clearDisplay();
  display.setCursor(0,0);
  display.println("📖 STRANDED BIBLE");
  display.println("================");
  
  for(int i=0; i<6; i++) {
    if(i == menuIndex) {
      display.print("> ");
    } else {
      display.print("  ");
    }
    display.println(mainMenu[i].name);
  }
  
  display.display();
}

void openFile(const char* filename) {
  if(strlen(filename) == 0) return;
  
  textFile = SD.open(filename);
  if(!textFile) {
    display.clearDisplay();
    display.setCursor(0,0);
    display.println("File not found");
    display.display();
    delay(2000);
    return;
  }
  
  // Read file into memory
  currentText = "";
  while(textFile.available()) {
    currentText += (char)textFile.read();
  }
  textFile.close();
  
  // Display first page
  scrollPosition = 0;
  showText();
  
  // Reading loop
  while(true) {
    if(digitalRead(BUTTON_UP) == LOW) {
      scrollPosition--;
      if(scrollPosition < 0) scrollPosition = 0;
      showText();
      delay(200);
    }
    
    if(digitalRead(BUTTON_DOWN) == LOW) {
      scrollPosition++;
      showText();
      delay(200);
    }
    
    if(digitalRead(BUTTON_SELECT) == LOW) {
      // Exit to menu
      delay(200);
      return;
    }
  }
}

void showText() {
  display.clearDisplay();
  display.setCursor(0,0);
  
  // Simple scrolling - just show lines from scrollPosition
  int line = 0;
  int startPos = 0;
  int endPos = 0;
  int currentLine = 0;
  
  // Find starting position
  for(int i=0; i<currentText.length(); i++) {
    if(currentText[i] == '\n') {
      if(currentLine >= scrollPosition) {
        endPos = i;
        break;
      }
      startPos = i+1;
      currentLine++;
    }
  }
  
  // Display 5 lines
  String toShow = currentText.substring(startPos);
  int linesShown = 0;
  int lastNewline = 0;
  
  for(int i=0; i<toShow.length() && linesShown < 5; i++) {
    if(toShow[i] == '\n') {
      display.println(toShow.substring(lastNewline, i));
      lastNewline = i+1;
      linesShown++;
    }
  }
  
  display.display();
}

        current_length = 0
        
        for word in words:
            if current_length + len(word) < 300:  # Rough chars per page

                current_length += len(word) + 1
            else:
                pages.append(' '.join(current_page))
                current_page = [word]

            pages.append(' '.join(current_page))

```
ARDUINO VERSION:
- Active current: 20mA
- Sleep current: 10μA
- Assume 1 hour reading per day
- Daily usage: 20mA × 1h = 20mAh
- Sleep: 0.01mA × 23h = 0.23mAh
- Total daily: 20.23mAh

CR2032 Battery: 220mAh
Life: 220 / 20.23 = 10.8 days continuous
With solar: indefinite

PI ZERO VERSION:
- Active: 120mA
- Sleep: 20mA (can't deep sleep easily)
- E-ink uses power only on refresh
- Daily 1 hour reading: 120mAh
- Sleep 23h @ 20mA = 460mAh
- Total daily: 580mAh

4000mAh battery life: 4000/580 = 6.9 days
With 5W solar: indefinite in sun
```
